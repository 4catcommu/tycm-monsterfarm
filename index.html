  match /databases/{database}/documents {

    match /monsterFarm/{doc} {
      allow read, write: if true;
    }

    match /monsterFarmPlayers/{playerId} {
      allow read, write: if true;
    }

      allow read, write: if true;

}

2. แก้ loginPlayer function

async function loginPlayer(){
  const name = document.getElementById('login-name').value.trim();
  const pin = document.getElementById('login-pin').value.trim();
  const status = document.getElementById('login-status');

  if(!name || !pin){
    status.textContent = "กรอกชื่อ + รหัสก่อน";
    return;
  }

  try{
    const data = await firebaseMonsterFarm.loginPlayer(name, pin, {
      savedProfile: {},
      inventory: []
    });

    playerSession = { key: firebaseMonsterFarm.currentPlayerKey, name };
    localStorage.setItem('tycm_current_player', JSON.stringify(playerSession));

    savedProfile = data.savedProfile || {};
    inventory = data.inventory || [];
    history = data.history || [];

    document.getElementById('login-ov').style.display = 'none';
    refreshLobby();

  }catch(err){
    status.textContent = err.message;
  }
}
